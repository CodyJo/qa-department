```python
"""Manual fix queue operations on the backlog.

Provides helpers to queue/dequeue findings and CLI entry point.
"""

import argparse
import json
import logging
import os

logger = logging.getLogger(__name__)

BACKLOG_PATH = os.path.join(
    os.path.dirname(os.path.dirname(os.path.abspath(__file__))),
    "dashboard", "backlog.json",
)


def _load_backlog(path=None):
    path = path or BACKLOG_PATH
    if not os.path.exists(path):
        return {"version": 1, "updated_at": "", "findings": {}}
    with open(path) as f:
        return json.load(f)


def _save_backlog(data, path=None):
    from datetime import datetime, timezone
    path = path or BACKLOG_PATH
    data["updated_at"] = datetime.now(timezone.utc).isoformat()
    os.makedirs(os.path.dirname(path) or ".", exist_ok=True)
    with open(path, "w") as f:
        json.dump(data, f, indent=2)


def queue_finding(hash_id, backlog_path=None):
    """Set a finding's status to 'queued'. Returns (success, message)."""
    backlog = _load_backlog(backlog_path)
    findings = backlog.get("findings", {})
    if hash_id not in findings:
        return False, f"Finding {hash_id} not found in backlog"
    entry = findings[hash_id]
    cf = entry.get("current_finding", {})
    if not cf.get("fixable_by_agent", False):
        return False, f"Finding {hash_id} is not fixable by agent"
    entry["status"] = "queued"
    _save_backlog(backlog, backlog_path)
    return True, f"Queued: {entry.get('title', hash_id)}"


def dequeue_finding(hash_id, backlog_path=None):
    """Set a finding's status back to 'open'. Returns (success, message)."""
    backlog = _load_backlog(backlog_path)
    findings = backlog.get("findings", {})
    if hash_id not in findings:
        return False, f"Finding {hash_id} not found in backlog"
    entry = findings[hash_id]
    entry["status"] = "open"
    _save_backlog(backlog, backlog_path)
    return True, f"Dequeued: {entry.get('title', hash_id)}"


def list_queued(backlog_path=None):
    """Return list of queued findings as dicts."""
    backlog = _load_backlog(backlog_path)
    findings = backlog.get("findings", {})
    queued = []
    for h, entry in findings.items():
        if entry.get("status") == "queued":
            queued.append({
                "hash": h,
                "repo": entry.get("repo", ""),
                "department": entry.get("department", ""),
                "severity": entry.get("severity", ""),
                "title": entry.get("title", ""),
                "effort": (entry.get("current_finding") or {}).get("effort", ""),
            })
    return queued


def clear_queue(backlog_path=None):
    """Reset all queued findings to open. Returns count cleared."""
    backlog = _load_backlog(backlog_path)
    findings = backlog.get("findings", {})
    count = 0
    for entry in findings.values():
        if entry.get("status") == "queued":
            entry["status"] = "open"
            count += 1
    if count > 0:
        _save_backlog(backlog, backlog_path)
    return count


def main(argv=None):
    """CLI entry point for queue subcommand."""
    parser = argparse.ArgumentParser(prog="python -m backoffice queue")
    sub = parser.add_subparsers(dest="action")
    add_p = sub.add_parser("add", help="Queue a finding for fix")
    add_p.add_argument("hash", help="16-char finding hash from backlog")
    rm_p = sub.add_parser("remove", help="Dequeue a finding")
    rm_p.add_argument("hash", help="16-char finding hash")
    sub.add_parser("list", help="List queued findings")
    sub.add_parser("clear", help="Clear the fix queue")

    args = parser.parse_args(argv)

    if args.action == "add":
        ok, msg = queue_finding(args.hash)
        print(msg)
        return 0 if ok else 1

    if args.action == "remove":
        ok, msg = dequeue_finding(args.hash)
        print(msg)
        return 0 if ok else 1

    if args.action == "list":
        items = list_queued()
        if not items:
            print("No findings queued for fix.")
            return 0
        print(f"{'Hash':<18s} {'Repo':<18s} {'Dept':<12s} {'Severity':<10s} Title")
        for item in items:
            print(f"{item['hash']:<18s} {item['repo']:<18s} {item['department']:<12s} {item['severity']:<10s} {item['title']}")
        print(f"\n{len(items)} finding{'s' if len(items) != 1 else ''} queued for fix")
        return 0

    if args.action == "clear":
        count = clear_queue()
        print(f"Cleared {count} finding{'s' if count != 1 else ''} from queue.")
        return 0

    parser.print_help()
    return 0
```

```bash
# ── Args ─────────────────────────────────────────────────────────────────────

TARGET_REPO=""
SYNC_TO_S3=false
RUN_DEPLOY=false
FINDINGS_OVERRIDE=""

while [ $# -gt 0 ]; do
  case "$1" in
    --sync)     SYNC_TO_S3=true ;;
    --deploy)   RUN_DEPLOY=true ;;
    --findings) shift; FINDINGS_OVERRIDE="${1:-}" ;;
    *)          [ -z "$TARGET_REPO" ] && TARGET_REPO="$1" ;;
  esac
  shift
done

if [ -z "$TARGET_REPO" ]; then
  echo "Usage: fix-bugs.sh /path/to/target-repo [--sync] [--deploy] [--findings <file>]" >&2
  exit 1
fi

if [ ! -d "$TARGET_REPO/.git" ]; then
  echo "Error: $TARGET_REPO is not a git repository" >&2
  exit 1
fi

REPO_NAME="$(basename "$TARGET_REPO")"
RESULTS_DIR="$QA_ROOT/results/$REPO_NAME"

if [ -n "$FINDINGS_OVERRIDE" ]; then
  FINDINGS_FILE="$RESULTS_DIR/$FINDINGS_OVERRIDE"
else
  FINDINGS_FILE="$RESULTS_DIR/findings.json"
fi
```

This replaces the `for arg` loop with a proper `while/shift` loop that can consume the next arg for `--findings`. The `TARGET_REPO` is now captured from the first non-flag argument instead of using `${1:?}`.

### Task 5: Create fix-queued.sh launcher

**Files:**
- Create: `scripts/fix-queued.sh`

- [ ] **Step 1: Write the script**

```bash
#!/usr/bin/env bash
# Back Office — Fix Queued Findings
# Usage: ./scripts/fix-queued.sh
#
# Reads dashboard/backlog.json for findings with status "queued",
# groups them by repo, and runs the fix agent on each group.

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
QA_ROOT="$(dirname "$SCRIPT_DIR")"
BACKLOG="$QA_ROOT/dashboard/backlog.json"

if [ ! -f "$BACKLOG" ]; then
  echo "No backlog file at $BACKLOG — run an audit first." >&2
  exit 1
fi

echo "╔══════════════════════════════════════════════════════════╗"
echo "║  Back Office — Fix Queued Findings                      ║"
echo "╚══════════════════════════════════════════════════════════╝"
echo ""

# Extract queued findings grouped by repo using Python
QUEUED_REPOS=$(python3 -c "
import json, sys
backlog = json.load(open('$BACKLOG'))
repos = {}
for h, entry in backlog.get('findings', {}).items():
    if entry.get('status') != 'queued':
        continue
    repo = entry.get('repo', '')
    if repo not in repos:
        repos[repo] = []
    cf = entry.get('current_finding', {})
    cf['_backlog_hash'] = h
    repos[repo].append(cf)
if not repos:
    print('EMPTY', file=sys.stderr)
    sys.exit(0)
for repo, findings in repos.items():
    print(f'{repo}:{len(findings)}')
" 2>&1) || true

if [ -z "$QUEUED_REPOS" ] || echo "$QUEUED_REPOS" | grep -q "EMPTY"; then
  echo "No findings queued for fix."
  exit 0
fi

echo "Queued findings by repo:"
echo "$QUEUED_REPOS" | while IFS=: read -r repo count; do
  echo "  $repo: $count findings"
done
echo ""

# Load targets config to resolve repo paths and check autonomy
TARGETS_FILE="$QA_ROOT/config/targets.yaml"

echo "$QUEUED_REPOS" | while IFS=: read -r repo count; do
  echo "── Fixing $repo ($count queued findings) ──"

  # Resolve repo path from targets.yaml
  REPO_PATH=$(python3 -c "
import yaml, sys
targets = yaml.safe_load(open('$TARGETS_FILE')).get('targets', [])
for t in targets:
    if t['name'] == '$repo':
        # Check autonomy policy
        autonomy = t.get('autonomy', {})
        if not autonomy.get('allow_fix', False):
            print('BLOCKED', file=sys.stderr)
            sys.exit(1)
        print(t['path'])
        sys.exit(0)
print('NOT_FOUND', file=sys.stderr)
sys.exit(1)
" 2>&1) || true

  if [ "$REPO_PATH" = "BLOCKED" ] || echo "$REPO_PATH" | grep -q "BLOCKED"; then
    echo "  Skipping $repo — allow_fix is false"
    echo ""
    continue
  fi

  if [ "$REPO_PATH" = "NOT_FOUND" ] || echo "$REPO_PATH" | grep -q "NOT_FOUND"; then
    echo "  Skipping $repo — not found in targets.yaml"
    echo ""
    continue
  fi

  RESULTS_DIR="$QA_ROOT/results/$repo"
  mkdir -p "$RESULTS_DIR"

  # Write queued findings as a temporary findings file
  python3 -c "
import json
backlog = json.load(open('$BACKLOG'))
findings = []
for h, entry in backlog.get('findings', {}).items():
    if entry.get('status') != 'queued' or entry.get('repo') != '$repo':
        continue
    cf = entry.get('current_finding', {})
    cf['_backlog_hash'] = h
    findings.append(cf)
output = {
    'repo_name': '$repo',
    'summary': {'total': len(findings)},
    'findings': findings
}
with open('$RESULTS_DIR/queued-findings.json', 'w') as f:
    json.dump(output, f, indent=2)
print(f'Wrote {len(findings)} findings to queued-findings.json')
"

  # Run fix agent with queued findings
  bash "$QA_ROOT/agents/fix-bugs.sh" "$REPO_PATH" --findings queued-findings.json && FIX_EXIT=0 || FIX_EXIT=$?

  # Update backlog statuses based on results
  python3 -c "
import json
backlog = json.load(open('$BACKLOG'))
fixes_path = '$RESULTS_DIR/fixes.json'
fix_map = {}
try:
    fixes = json.load(open(fixes_path))
    for fix in fixes.get('fixes', []):
        fix_map[fix.get('finding_id', '')] = fix.get('status', 'failed')
except (FileNotFoundError, json.JSONDecodeError):
    pass

status = 'fixed' if $FIX_EXIT == 0 and fix_map else 'fix_failed'
updated = 0
for h, entry in backlog.get('findings', {}).items():
    if entry.get('status') != 'queued' or entry.get('repo') != '$repo':
        continue
    fid = (entry.get('current_finding') or {}).get('id', '')
    if fid in fix_map and fix_map[fid] == 'fixed':
        entry['status'] = 'fixed'
    else:
        entry['status'] = 'fix_failed' if $FIX_EXIT != 0 or fid in fix_map else 'fix_failed'
    updated += 1

from datetime import datetime, timezone
backlog['updated_at'] = datetime.now(timezone.utc).isoformat()
with open('$BACKLOG', 'w') as f:
    json.dump(backlog, f, indent=2)
print(f'Updated {updated} backlog entries')
"

  echo ""
done

# Refresh dashboard
echo "Refreshing dashboard data..."
python3 -m backoffice refresh 2>/dev/null || true
echo "Done."
```
